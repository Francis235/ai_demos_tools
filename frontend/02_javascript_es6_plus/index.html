🔤 Variables & Declarations

let vs const vs var

Learn the differences between modern variable declarations

// const for constants
const PI = 3.14159;

// let for variables
let userName = 'Alice';

// var is old school
var age = 25; // avoid this!

Template Literals

String interpolation and multiline strings

const name = 'World';
const greeting = `Hello, ${name}!
Welcome to JavaScript ES6+`;

console.log(greeting);

Destructuring

Extract values from arrays and objects

const [a, b] = [1, 2];
const {name, age} = user;

// With default values
const [x = 10] = [];

⚡ Modern Functions

Arrow Functions

Concise function syntax and lexical this

// Traditional function
function add(a, b) {
  return a + b;
}

// Arrow function
const add = (a, b) => a + b;

Default Parameters

Set default values for function parameters

function greet(name = 'Friend') {
  return `Hello, ${name}!`;
}

const multiply = (a, b = 1) => a * b;

Rest & Spread

Handle variable number of arguments

// Rest parameters
const sum = (...numbers) => 
  numbers.reduce((a, b) => a + b, 0);

// Spread operator
const arr = [1, 2, 3];
const newArr = [...arr, 4, 5];

📊 Array Methods

Map, Filter, Reduce

Functional programming with arrays

const numbers = [1, 2, 3, 4, 5];

const doubled = numbers.map(x => x * 2);
const evens = numbers.filter(x => x % 2 === 0);
const sum = numbers.reduce((a, b) => a + b);

Find & Some/Every

Search and validate array elements

const users = [{name: 'Alice', age: 25}];

const user = users.find(u => u.name === 'Alice');
const hasAdults = users.some(u => u.age >= 18);
const allAdults = users.every(u => u.age >= 18);

Array Destructuring

Extract array elements efficiently

const colors = ['red', 'green', 'blue'];
const [first, second, ...rest] = colors;

// Swapping variables
let a = 1, b = 2;
[a, b] = [b, a];

🔧 Object Features

Object Destructuring

Extract object properties elegantly

const person = {name: 'Alice', age: 30, city: 'NYC'};

const {name, age} = person;
const {name: fullName} = person; // renaming
const {country = 'USA'} = person; // default

Object Methods

Object.keys, values, entries and more

const obj = {a: 1, b: 2, c: 3};

const keys = Object.keys(obj);
const values = Object.values(obj);
const entries = Object.entries(obj);
const merged = Object.assign({}, obj, {d: 4});

Enhanced Object Literals

Shorthand properties and computed keys

const name = 'Alice';
const age = 30;

// Shorthand properties
const person = {name, age};

// Computed property names
const key = 'dynamic';
const obj = {[key]: 'value'};

⏱️ Async Programming

Promises

Handle asynchronous operations elegantly

const fetchData = () => {
  return new Promise((resolve, reject) => {
    setTimeout(() => resolve('Data loaded!'), 1000);
  });
};

fetchData().then(data => console.log(data));

Async/Await

Write asynchronous code that looks synchronous

async function loadData() {
  try {
    const data = await fetchData();
    console.log(data);
  } catch (error) {
    console.error(error);
  }
}

Fetch API

Modern way to make HTTP requests

// GET request
const response = await fetch('/api/users');
const users = await response.json();

// POST request  
await fetch('/api/users', {
  method: 'POST',
  body: JSON.stringify(user)
});

📦 ES6 Modules

Import/Export

Modular JavaScript with imports and exports

// math.js
export const add = (a, b) => a + b;
export default function multiply(a, b) {
  return a * b;
}

// main.js
import multiply, {add} from './math.js';

Dynamic Imports

Load modules on demand

// Lazy loading modules
async function loadModule() {
  const module = await import('./utils.js');
  module.doSomething();
}

// Conditional imports
if (condition) {
  const {feature} = await import('./feature.js');
}

Classes

Object-oriented programming in JavaScript

class Animal {
  constructor(name) {
    this.name = name;
  }
  
  speak() {
    console.log(`${this.name} makes a sound`);
  }
}

class Dog extends Animal {
  speak() {
    console.log(`${this.name} barks`);
  }
}
